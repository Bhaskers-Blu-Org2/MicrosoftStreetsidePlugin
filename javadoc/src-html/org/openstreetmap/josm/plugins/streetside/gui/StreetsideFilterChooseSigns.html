002package org.openstreetmap.josm.plugins.streetside.gui;
003
004import static org.openstreetmap.josm.tools.I18n.tr;
005import static org.openstreetmap.josm.tools.I18n.trc;
006
007import java.awt.Dimension;
008
009import javax.swing.JCheckBox;
010import javax.swing.JLabel;
011import javax.swing.JPanel;
012
013import org.openstreetmap.josm.tools.ImageProvider;
014
015/**
016 * UI to choose which signs must be displayed.
017 *
018 * @author nokutu
019 *
020 */
021public final class StreetsideFilterChooseSigns extends JPanel {
022
023  private static final long serialVersionUID = -4858656447066921229L;
024
025  /** Max speed signs */
026  public final JCheckBox maxSpeed = new JCheckBox();
027  /** Stop signs */
028  public final JCheckBox stop = new JCheckBox();
029  /** Give way signs */
030  public final JCheckBox giveWay = new JCheckBox();
031  /** Roundabout sings */
032  public final JCheckBox roundabout = new JCheckBox();
033  /** Forbidden access or forbidden direction signs */
034  public final JCheckBox access = new JCheckBox();
035  /** Intersection danger signs */
036  public final JCheckBox intersection = new JCheckBox();
037  /** Mandatory direction signs */
038  public final JCheckBox direction = new JCheckBox();
039  /** Uneven pavement signs */
040  public final JCheckBox uneven = new JCheckBox();
041  /** No parking signs */
042  public final JCheckBox noParking = new JCheckBox();
043  /** Forbidden overtaking signs */
044  public final JCheckBox noOvertaking = new JCheckBox();
045  /** Pedestrian crossing signs */
046  public final JCheckBox crossing = new JCheckBox();
047  /** Forbidden turn signs */
048  public final JCheckBox noTurn = new JCheckBox();
049
050  /**
051   * The {@link JCheckBox}es where the respective tag should be searched
052   */
053  public final JCheckBox[] signCheckboxes = new JCheckBox[]{maxSpeed, stop, giveWay, roundabout, access, intersection, direction, uneven, noParking, noOvertaking, crossing, noTurn};
054  public static final String[] SIGN_TAGS = {
055    "(complementary)|(regulatory)--maximum-speed-limit", "(regulatory)|(warning)--stop",
056    "(regulatory)|(warning)--yield", "(warning)|(regulatory)--roundabout", "regulatory--no-entry",
057    "warning--(crossroads)|(junction)", "regulatory--(turn)|(go-straight)", "warning--(uneven)|(slippery)",
058    "regulatory--no-parking", "regulatory--no-overtaking", "information--pedestrians-crossing", "regulatory--no-.+-turn"
059  };
060
061  private static StreetsideFilterChooseSigns instance;
062
063  private StreetsideFilterChooseSigns() {
064    // i18n: traffic sign
065    addCheckBoxWithLabel(this, maxSpeed, true, "signs/speed.png", tr("Speed limit"));
066    // i18n: traffic sign
067    addCheckBoxWithLabel(this, stop, true, "signs/stop.png", trc("name of the traffic sign", "Stop"));
068    // i18n: traffic sign
069    addCheckBoxWithLabel(this, giveWay, true, "signs/right_of_way.png", tr("Give way"));
070    // i18n: traffic sign
071    addCheckBoxWithLabel(this, roundabout, true, "signs/roundabout_right.png", tr("Roundabout"));
072    // i18n: traffic sign
073    addCheckBoxWithLabel(this, access, true, "signs/no_entry.png", tr("No entry"));
074    // i18n: traffic sign
075    addCheckBoxWithLabel(this, intersection, true, "signs/intersection_danger.png", tr("Intersection danger"));
076    // i18n: traffic sign
077    addCheckBoxWithLabel(this, direction, true, "signs/only_straight_on.png", tr("Mandatory direction (any)"));
078    // i18n: traffic sign
079    addCheckBoxWithLabel(this, uneven, true, "signs/uneven.png", tr("Uneven road"));
080    // i18n: traffic sign
081    addCheckBoxWithLabel(this, noParking, true, "signs/no_parking.png", tr("No parking"));
082    // i18n: traffic sign
083    addCheckBoxWithLabel(this, noOvertaking, true, "signs/no_overtaking.png", tr("No overtaking"));
084    // i18n: traffic sign
085    addCheckBoxWithLabel(this, crossing, true, "signs/crossing.png", tr("Pedestrian crossing"));
086    // i18n: traffic sign
087    addCheckBoxWithLabel(this, noTurn, true, "signs/no_turn.png", tr("No turn"));
088
089    setPreferredSize(new Dimension(600, 250));
090  }
091
092  private static void addCheckBoxWithLabel(final JPanel parentPanel, final JCheckBox checkBox,
093      final boolean isSelected, final String iconPath, final String labelText) {
094    final JPanel checkBoxPanel = new JPanel();
095    final JLabel checkBoxLabel = new JLabel(labelText);
096
097    checkBoxLabel.setIcon(new ImageProvider(iconPath).get());
098    checkBoxLabel.setLabelFor(checkBox);
099    checkBoxPanel.add(checkBoxLabel);
100    checkBox.setSelected(isSelected);
101    checkBoxPanel.add(checkBox);
102    parentPanel.add(checkBoxPanel);
103  }
104
105  /**
106   * Return the unique instance of the class.
107   *
108   * @return the unique instance of the class.
109   */
110  public static synchronized StreetsideFilterChooseSigns getInstance() {
111    if (instance == null) {
112      instance = new StreetsideFilterChooseSigns();
113    }
114    return instance;
115  }
116}
