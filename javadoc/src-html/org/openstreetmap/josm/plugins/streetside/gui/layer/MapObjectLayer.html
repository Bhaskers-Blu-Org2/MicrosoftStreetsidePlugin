035  public enum STATUS {
036    DOWNLOADING(I18n.marktr("Downloading map objectsâ€¦"), Color.YELLOW),
037    COMPLETE(I18n.marktr("All map objects loaded."), Color.GREEN),
038    INCOMPLETE(I18n.marktr("Too many map objects, zoom in to see all."), Color.ORANGE),
039    FAILED(I18n.marktr("Downloading map objects failed!"), Color.RED);
040
041    private final int colorValue;
042    public final String message;
043
044    STATUS(final String message, final Color color) {
045      colorValue = color.getRGB();
046      this.message = message;
047    }
048
049    /**
050     * Note: The color is stored as int and each time returned as new {@link Color} instance,
051     * because the class {@link Color} is not strictly immutable.
052     * @return the color associated with the current status
053     */
054    public Color getColor() {
055      return new Color(colorValue, true);
056    }
057  }
058
059  private static MapObjectLayer instance;
060
061  private STATUS status = STATUS.COMPLETE;
062  private MapObjectDownloadRunnable downloadRunnable;
063  private MapObjectDownloadRunnable nextDownloadRunnable;
064  private final Collection<MapObject> objects = new HashSet<>();
065
066  private final Map<String, ImageIcon> scaledIcons = new HashMap<>();
067
068  private MapObjectLayer() {
069    super(I18n.tr("Display Mapillary street signs"));
070    NavigatableComponent.addZoomChangeListener(this);
071    StreetsideProperties.MAPOBJECT_ICON_SIZE.addListener(val -> {
072      scaledIcons.clear();
073      finishDownload(false);
074    });
075    zoomChanged();
076  }
077
078  private static void clearInstance() {
079    synchronized (MapObjectLayer.class) {
080      instance = null;
081    }
082  }
083
084  public static MapObjectLayer getInstance() {
085    synchronized (MapObjectLayer.class) {
086      if (instance == null) {
087        instance = new MapObjectLayer();
088      }
089      return instance;
090    }
091  }
092
093  public boolean isDownloadRunnableScheduled() {
094    synchronized (this) {
095      return nextDownloadRunnable != null;
096    }
097  }
098
099  public void finishDownload(boolean replaceMapObjects) {
100    synchronized (this) {
101      final MapObjectDownloadRunnable currentRunnable = downloadRunnable;
102      if (currentRunnable != null) {
103        synchronized (objects) {
104          if (replaceMapObjects) {
105            objects.clear();
106          }
107          objects.addAll(currentRunnable.getMapObjects());
108        }
109      }
110      downloadRunnable = null;
111      if (nextDownloadRunnable != null) {
112        downloadRunnable = nextDownloadRunnable;
113        nextDownloadRunnable = null;
114        new Thread(downloadRunnable, "downloadMapObjects").start();
115      }
116    }
117    new Thread(() -> {
118      synchronized (objects) {
119        for (MapObject object : objects) {
120          if (!scaledIcons.containsKey(object.getValue())) {
121            scaledIcons.put(
122              object.getValue(),
123              ImageUtil.scaleImageIcon(MapObject.getIcon(object.getValue()), StreetsideProperties.MAPOBJECT_ICON_SIZE.get())
124            );
125          }
126        }
127      }
128      invalidate();
129    }, "downloadMapObjectIcons").start();
130  }
131
132  public int getObjectCount() {
133    return objects.size();
134  }
135
136  public void setStatus(STATUS status) {
137    this.status = status;
138    invalidate();
139  }

141  @Override
142  public void paint(Graphics2D g, MapView mv, Bounds bbox) {
143    final long startTime = System.currentTimeMillis();
144    final Collection<MapObject> displayedObjects = new HashSet<>();
145    synchronized (this) {
146      final MapObjectDownloadRunnable currentRunnable = downloadRunnable;
147      if (currentRunnable != null) {
148        displayedObjects.addAll(currentRunnable.getMapObjects());
149      }
150    }
151    displayedObjects.addAll(objects);
152
153    for (MapObject object : displayedObjects) {
154      final ImageIcon icon = scaledIcons.get(object.getValue());
155      if (icon != null) {
156        final Point p = mv.getPoint(object.getCoordinate());
157        g.drawImage(
158          icon.getImage(),
159          p.x - icon.getIconWidth() / 2,
160          p.y - icon.getIconHeight() / 2,
161          null
162        );
163      }
164    }
165
166    final STATUS currentStatus = status;
167    g.setFont(g.getFont().deriveFont(Font.PLAIN).deriveFont(12f));
168    g.setColor(currentStatus.getColor());
169    final FontMetrics fm = g.getFontMetrics();
170    g.fillRect(0, mv.getHeight() - fm.getAscent() - fm.getDescent(), fm.stringWidth(I18n.tr(currentStatus.message)), fm.getAscent() + fm.getDescent());
171    g.setColor(Color.BLACK);
172    g.drawString(I18n.tr(currentStatus.message), 0, mv.getHeight() - fm.getDescent());
173    Logging.debug("{0} painted in {1} milliseconds.", MapObjectLayer.class.getName(), System.currentTimeMillis() - startTime);
174  }

211  public void scheduleDownload(final Bounds bounds) {
212    synchronized(this) {
213      if (downloadRunnable == null) {
214        downloadRunnable = new MapObjectDownloadRunnable(this, bounds);
215        new Thread(downloadRunnable).start();
216      } else {
217        nextDownloadRunnable = new MapObjectDownloadRunnable(this, bounds);
218      }
219    }
220  }

223   * @see org.openstreetmap.josm.gui.layer.Layer#destroy()
224   */

228    super.destroy();
229  }

231  @Override
232  public void zoomChanged() {
233    MapView mv = StreetsidePlugin.getMapView();
234    if (mv != null) {
235      scheduleDownload(mv.getState().getViewArea().getLatLonBoundsBox());
236    }
