166  /**
167   * Applies the selected filter.
168   */
169  public synchronized void refresh() {
170    final boolean layerVisible = StreetsideLayer.hasInstance() && StreetsideLayer.getInstance().isVisible();
171    final boolean imported = this.imported.isSelected();
172    final boolean downloaded = this.downloaded.isSelected();
173    final boolean timeFilter = filterByDateCheckbox.isSelected();
174    final boolean onlySigns = this.onlySigns.isSelected();
175
176    // This predicate returns true is the image should be made invisible
177    Predicate<StreetsideAbstractImage> shouldHide =
178      img -> {
179        if (!layerVisible) {
180          return true;
181        }
182        if (timeFilter && checkValidTime(img)) {
183          return true;
184        }
185        if (!imported && img instanceof StreetsideImportedImage) {
186          return true;
187        }
188        if (img instanceof StreetsideImage) {
189          if (!downloaded) {
190            return true;
191          }
192          if (onlySigns && (((StreetsideImage) img).getDetections().isEmpty() || !checkSigns((StreetsideImage) img))) {
193            return true;
194          }
195          UserProfile userProfile = ((StreetsideImage) img).getUser();
196          if (!"".equals(user.getText()) && (userProfile == null || !user.getText().equals(userProfile.getUsername()))) {
197            return true;
198          }
199        }
200        return false;
201      };
202
203    if (StreetsideLayer.hasInstance()) {
204      StreetsideLayer.getInstance().getData().getImages().parallelStream().forEach(img -> img.setVisible(!shouldHide.test(img)));
205    }
206
207    StreetsideLayer.invalidateInstance();
208  }
209
210  private boolean checkValidTime(StreetsideAbstractImage img) {
211    Long currentTime = currentTime();
212    for (int i = 0; i < 3; i++) {
213      if (TIME_LIST[i].equals(time.getSelectedItem()) &&
214        //img.getCapturedAt() < currentTime - spinnerModel.getNumber().doubleValue() * TIME_FACTOR[i]) {
215        img.getCd() < currentTime - spinnerModel.getNumber().doubleValue() * TIME_FACTOR[i]) {
216        return true;
217      }
218    }
219    return false;
220  }
221
222  /**
223   * Checks if the image fulfills the sign conditions.
224   *
225   * @param img The {@link StreetsideAbstractImage} object that is going to be
226   * checked.
227   *
228   * @return {@code true} if it fulfills the conditions; {@code false}
229   * otherwise.
230   */
231  private static boolean checkSigns(StreetsideImage img) {
232    for (int i = 0; i < StreetsideFilterChooseSigns.SIGN_TAGS.length; i++) {
233      if (checkSign(img, StreetsideFilterChooseSigns.getInstance().signCheckboxes[i], StreetsideFilterChooseSigns.SIGN_TAGS[i]))
234        return true;
235    }
236    return false;
237  }
238
239  private static boolean checkSign(StreetsideImage img, JCheckBox signCheckBox, String signTag) {
240    boolean contains = false;
241    for (ImageDetection detection : img.getDetections()) {
242      if (Pattern.compile(signTag).matcher(detection.getValue()).find()) {
243        contains = true;
244      }
245    }
246    return contains == signCheckBox.isSelected() && contains;
247  }
248
249  private static long currentTime() {
250    Calendar cal = Calendar.getInstance();
251    return cal.getTimeInMillis();
252  }

333      final JOptionPane pane = new JOptionPane(
334        StreetsideFilterChooseSigns.getInstance(),
335        JOptionPane.PLAIN_MESSAGE,
336        JOptionPane.OK_CANCEL_OPTION
337      );
338      JDialog dlg = pane.createDialog(Main.parent, tr("Choose signs"));
