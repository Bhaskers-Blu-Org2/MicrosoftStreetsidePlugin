002package org.openstreetmap.josm.plugins.streetside;
003
004import java.awt.image.BufferedImage;
005import java.util.EnumSet;
006import java.util.HashMap;
007import java.util.List;
008import java.util.Map;
009
010import org.openstreetmap.josm.data.coor.LatLon;
011import org.openstreetmap.josm.plugins.streetside.cubemap.CubemapUtils;
012import org.openstreetmap.josm.plugins.streetside.utils.StreetsideProperties;
013
014public class StreetsideCubemap extends StreetsideAbstractImage implements Comparable<StreetsideAbstractImage>{
015
016        private static Map<String,Map<String,BufferedImage>> face2TilesMap = new HashMap<String,Map<String,BufferedImage>>();
017
018        /**
019        * If two values for field cd differ by less than EPSILON both values are considered equal.
020        */
021        private static final float EPSILON = 1e-5f;
022
023
024        /**
025           * Main constructor of the class StreetsideCubemap
026           *
027           * @param key    The unique identifier of the cubemap.
028           * @param latLon The latitude and longitude where it is positioned.
029           * @param cd     The direction of the images in degrees, meaning 0 north.
030           */
031
032          public StreetsideCubemap(String quadId, LatLon latLon, double ca) {
033            super(quadId, latLon, ca);
034            face2TilesMap = new HashMap();
035
036            EnumSet.allOf(CubemapUtils.CubemapFaces.class).forEach(face -> {
037                        face2TilesMap.put(face.getValue(),new HashMap<String,BufferedImage>());
038                });
039
040          }
041
042          public StreetsideCubemap(String quadId, LatLon latLon) {
043            super(quadId, latLon, 0.0);
044          }
045
046          // Default constructor for Jackson/JSON Deserializattion
047          public StreetsideCubemap(String quadId, LatLon latlon, List<StreetsideImage> tiles, double he) {
048            super(quadId, latlon, he);
049          }
050
051        /**
052         * @return the face2TilesMap
053         */
054        public Map<String, Map<String,BufferedImage>> getFace2TilesMap() {
055                return face2TilesMap;
056        }
057
058    @Override
059          public int compareTo(StreetsideAbstractImage image) {
060            if (image instanceof StreetsideImage) {
061              return id.compareTo(((StreetsideImage) image).getId());
062            }
063            return hashCode() - image.hashCode();
064          }
065
066          @Override
067          public int hashCode() {
068            return id.hashCode();
069          }
070
071          @Override
072          public void stopMoving() {
073            super.stopMoving();
074          }
075
076          @Override
077          public void turn(double ca) {
078            super.turn(ca);
079          }
080
081        public int getHeight() {
082                return StreetsideProperties.SHOW_HIGH_RES_STREETSIDE_IMAGERY.get().booleanValue()?1016:510;
083        }
084
085        public void resetFaces2TileMap() {
086                face2TilesMap = new HashMap<>();
087
088                EnumSet.allOf(CubemapUtils.CubemapFaces.class).forEach(face -> {
089                        face2TilesMap.put(face.getValue(), new HashMap<String, BufferedImage>());
090                });
091        }
092}
