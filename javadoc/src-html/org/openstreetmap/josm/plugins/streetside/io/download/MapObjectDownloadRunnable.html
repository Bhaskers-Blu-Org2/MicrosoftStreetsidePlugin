002package org.openstreetmap.josm.plugins.streetside.io.download;
003
004import java.io.BufferedInputStream;
005import java.io.IOException;
006import java.net.URL;
007import java.net.URLConnection;
008import java.util.Collection;
009import java.util.HashSet;
010import java.util.function.Function;
011
012import javax.json.Json;
013import javax.json.JsonException;
014import javax.json.JsonReader;
015
016import org.openstreetmap.josm.data.Bounds;
017import org.openstreetmap.josm.gui.Notification;
018import org.openstreetmap.josm.plugins.streetside.StreetsidePlugin;
019import org.openstreetmap.josm.plugins.streetside.gui.layer.MapObjectLayer;
020import org.openstreetmap.josm.plugins.streetside.gui.layer.MapObjectLayer.STATUS;
021import org.openstreetmap.josm.plugins.streetside.model.MapObject;
022import org.openstreetmap.josm.plugins.streetside.utils.StreetsideProperties;
023import org.openstreetmap.josm.plugins.streetside.utils.StreetsideURL.APIv3;
024import org.openstreetmap.josm.plugins.streetside.utils.api.JsonDecoder;
025import org.openstreetmap.josm.plugins.streetside.utils.api.JsonMapObjectDecoder;
026import org.openstreetmap.josm.tools.I18n;
027import org.openstreetmap.josm.tools.ImageProvider.ImageSizes;
028import org.openstreetmap.josm.tools.Logging;
029
030public class MapObjectDownloadRunnable implements Runnable {
031  private final Bounds bounds;
032  private final MapObjectLayer layer;
033  private static final Function<Bounds, URL> URL_GEN = APIv3::searchMapObjects;
034
035  private final Collection<MapObject> result = new HashSet<>();
036
037  public MapObjectDownloadRunnable(final MapObjectLayer layer, final Bounds bounds) {
038    this.bounds = bounds;
039    this.layer = layer;
040  }
041
042  public Collection<MapObject> getMapObjects() {
043    return result;
044  }
045
046  @Override
047  public void run() {
048    layer.setStatus(MapObjectLayer.STATUS.DOWNLOADING);
049    URL nextURL = URL_GEN.apply(bounds);
050    try {
051      while (nextURL != null && result.size() < StreetsideProperties.MAX_MAPOBJECTS.get() && !layer.isDownloadRunnableScheduled()) {
052        final int prevResultSize = result.size();
053        final long startTime = System.currentTimeMillis();
054        final URLConnection con = nextURL.openConnection();
055        try (JsonReader reader = Json.createReader(new BufferedInputStream(con.getInputStream()))) {
056          result.addAll(JsonDecoder.decodeFeatureCollection(
057            reader.readObject(),
058            JsonMapObjectDecoder::decodeMapObject
059          ));
060        }
061        layer.invalidate();
062        BoundsDownloadRunnable.logConnectionInfo(con, String.format(
063          "%d map objects in %.2f s",
064          result.size() - prevResultSize,
065          (System.currentTimeMillis() - startTime) / 1000f
066        ));
067        nextURL = APIv3.parseNextFromLinkHeaderValue(con.getHeaderField("Link"));
068      }
069    } catch (IOException | JsonException e) {
070      String message = I18n.tr("{0}\nCould not read map objects from URL\n{1}!", e.getLocalizedMessage(), nextURL.toString());
071      Logging.log(Logging.LEVEL_WARN, message, e);
072      new Notification(message)
073        .setIcon(StreetsidePlugin.LOGO.setSize(ImageSizes.LARGEICON).get())
074        .setDuration(Notification.TIME_LONG)
075        .show();
076      layer.setStatus(STATUS.FAILED);
077      layer.finishDownload(false);
078      return;
079    }
080    if (nextURL == null) {
081      layer.setStatus(STATUS.COMPLETE);
082    } else {
083      layer.setStatus(result.size() >= StreetsideProperties.MAX_MAPOBJECTS.get() ? STATUS.INCOMPLETE : STATUS.DOWNLOADING);
084    }
085    layer.finishDownload(nextURL == null || result.size() >= StreetsideProperties.MAX_MAPOBJECTS.get());
086    try {
087      Thread.sleep(1000); // Buffer between downloads to avoid too many downloads when e.g. panning around
088    } catch (InterruptedException e) {
089      Logging.debug(e);
090      Thread.currentThread().interrupt();
091    }
092  }
093
094}
