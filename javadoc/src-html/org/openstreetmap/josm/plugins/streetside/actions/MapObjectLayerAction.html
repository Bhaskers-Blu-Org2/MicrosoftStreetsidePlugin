002package org.openstreetmap.josm.plugins.streetside.actions;
003
004import java.awt.event.ActionEvent;
005
006import org.openstreetmap.josm.actions.JosmAction;
007import org.openstreetmap.josm.gui.MainApplication;
008import org.openstreetmap.josm.gui.util.GuiHelper;
009import org.openstreetmap.josm.plugins.streetside.StreetsidePlugin;
010import org.openstreetmap.josm.plugins.streetside.gui.layer.MapObjectLayer;
011import org.openstreetmap.josm.tools.I18n;
012import org.openstreetmap.josm.tools.ImageProvider.ImageSizes;
013
014public class MapObjectLayerAction extends JosmAction {
015
016  private static final long serialVersionUID = 302786746122185446L;
017
018
019  public MapObjectLayerAction() {
020    super(
021      I18n.tr("Mapillary object layer"),
022      StreetsidePlugin.LOGO.setSize(ImageSizes.DEFAULT),
023      I18n.tr("Displays the layer displaying the map objects detected by the Bing API"),
024      null,
025      false,
026      "streetsideObjectLayer",
027      false
028    );
029  }
030
031
032  @Override
033  public void actionPerformed(ActionEvent e) {
034    GuiHelper.runInEDTAndWait(() -> {
035      // Synchronization lock must be held by EDT thread
036      // See {@link LayerManager#addLayer(org.openstreetmap.josm.gui.layer.Layer, boolean)}.
037      synchronized (MainApplication.getLayerManager()) {
038        if (!MainApplication.getLayerManager().containsLayer(MapObjectLayer.getInstance())) {
039          MainApplication.getLayerManager().addLayer(MapObjectLayer.getInstance());
040        }
041      }
042    });
043  }
044}
